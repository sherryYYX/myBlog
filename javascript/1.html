# Set 集合：表示无重复值的有序列表

let set = new Set(); 

// 添加元素
set.add(1,2) // 添加的内容可以是任意类型数据，也可以是集合

// 删除元素
set.delete(2)

// 校验某值是否在集合中
const result = set.has(2) //result : true

// 集合的长度
set.size()

//
set.forEach((value,key)=>{
    console.log(value,key)
})

// 将set转换成数组
let set2 = new Set([1,2,2,3,3,4])
console.log(set2) // {1,2,3,4} 这不是想要的结果
// 使用扩展运算符
let arr = [...set2]
console.log(arr) // [1,2,3,4]

// 1.set中对象的引用无法被释放
let set3 = new Set(), obj = {};
set3.add(obj);

// 释放当前资源
obj = null; 
console.log(set3) //当前对象无法被释放

// 所以使用弱应用
let set4 = new WeakSet(), obj = {};
set4.add(obj);
// 释放当前资源
obj = null; 
console.log(set4) //Entries 里没有obj

//WeakSet
// 1.不能传入非对象类型的参数
// 2.不可迭代，遍历
// 3. 没有forEach()
// 4. 没有 size 属性

# Map

Map 类型是键值对的有序列表，键和值是任意类型

let map = new Map()

//设置键和值
map.set('name','张三') 
console.log(map) //{"name"=>"张三"}

// 获取值
console.log(map.get('name')) // 张三

// 判断是否有某元素
map.has('name') //true

//
map.set(['a',[1,2,3]],'hello')
console.log(map) // {Array(2) => "hello"}
