## 两个数组的处理

::: details 示例
``` javaScript
let LongList = [
    {
      index: 0,
      url:'https://upload.i-cbao.com/files/20210526/b87e2868-4627-410b-b070-feb24f3566f9.jpeg',
      isDelete: false,
      remark: "",
      status: 1,
      subType: 1,
      title: "左后45°带有完整车牌号照片(含车顶)",
      type: 0,
    },
    {
      index: 1,
      url:'https://upload.i-cbao.com/files/20210526/b87e2868-4627-410b-b070-feb24f3566f9.jpeg',
      isDelete: false,
      remark: "",
      status: 1,
      subType: 1,
      title: "左后45°带有完整车牌号照片(含车顶)",
      type: 0,
    },
  ]
  
  let shortList =  [
    {
      index: 0,
      url:'https://upload.i-cbao.com/files/20210526/b87e2868-4627-410b-b070-feb24f3566f9.jpeg',
      isDelete: false,
      remark: "",
      status: 1,
      subType: 1,
      title: "左后45°带有完整车牌号照片(含车顶)",
      type: 0,
    },
  ]
  
  /**
   * 两个长度不同数组，依据长的为基准，两个数组相互对应，少的项值为空（工具函数）
   * @param {数组1} arr1List 
   * @param {数组2} arr2List 
   */
  function tool(arr1List,arr2List){
    arr1List.length > arr2List.length ? [arr1List,arr2List] = [arr2List,arr1List]: ''
   arr2List.map(arr2 =>{
     // 从少的数组中找到有多的对应的
   let targetItem =   arr1List.find(arr1=> arr1.index ===arr2.index  )
    // 如果多的item 没有， 就给少的加上‘’
    targetItem ? '': arr1List.push({...arr2,url:''}) 
  })
  }
  tool(LongList,shortList)
    console.log(shortList);


  /**
   * 两个长度不同数组，依据短的为基准，两个数组相互对应，少的项值为空（工具函数）
   * @param {数组1} arr1List 
   * @param {数组2} arr2List 
   */
  function tool2(arr1List,arr2List){
    let result = []
    arr1List.length > arr2List.length ? [arr1List,arr2List] = [arr2List,arr1List]: ''
   arr2List.filter(arr2 =>{
   let targetItem =   arr1List.find(arr1=> arr1.index ===arr2.index  )  
  targetItem? result.push(targetItem):''
  })
  return  result
  }
  LongList = tool2(LongList,shortList)
    console.log(LongList);

```
