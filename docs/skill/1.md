## 生成Select选项数据的方法
工作中经常遇到枚举值的下拉框和数据展示，此方法可以清晰的对枚举数据组的处理

::: details 示例展示
```javaScript
const VehicleStatus = {
    //批单申请
    EndorsementApply: -4,
    //余额不足
    BalanceNotEnough: -3,
    //待完善资料
    Incomplete: -2,
  };
  
  const VehicleStatusList = [
    //批单申请
    -4,
    //余额不足
    -3,
    //待完善资料
   -2,
  ];
  
   function getVehicleStatusDropList(VehicleStatusList) {
    const result= [];
    VehicleStatusList.forEach(status => {
      result.push({ value: status, name: display(status) });
    });
    return result;
  }

  
  function display(status) {
    switch (status) {
      case VehicleStatus.EndorsementApply:
        return '批单申请';
        break;
      case VehicleStatus.BalanceNotEnough:
        return '余额不足';
        break;
      case VehicleStatus.Incomplete:
        return '待完善资料';
        break;
      default:
        break;
    }
  }

  console.log(getVehicleStatusDropList(VehicleStatusList));
  
  // [
  //   { value: -4, name: '批单申请' },
  //   { value: -3, name: '余额不足' },
  //   { value: -2, name: '待完善资料' }
  // ]
```
